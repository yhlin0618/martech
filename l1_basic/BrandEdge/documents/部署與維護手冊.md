#### 健康檢查端點
```r
# 在 app.R 中添加健康檢查
observeEvent(input$health_check, {
  tryCatch({
    # 檢查資料庫連線
    con <- get_con()
    dbGetQuery(con, "SELECT 1")
    dbDisconnect(con)
    
    # 檢查 API
    api_key <- Sys.getenv("OPENAI_API_KEY")
    api_status <- !is.null(api_key) && api_key != ""
    
    output$health_status <- renderText({
      toJSON(list(
        status = "healthy",
        database = "connected",
        api = if(api_status) "configured" else "missing",
        timestamp = Sys.time()
      ))
    })
  }, error = function(e) {
    output$health_status <- renderText({
      toJSON(list(
        status = "unhealthy",
        error = e$message,
        timestamp = Sys.time()
      ))
    })
  })
})
```

#!/bin/bash
# monitor.sh

while true; do
  # CPU 使用率
  CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
  
  # 記憶體使用
  MEM=$(free -m | awk 'NR==2{printf "%.2f", $3*100/$2}')
  
  # 磁碟使用
  DISK=$(df -h / | awk 'NR==2{print $5}')
  
  # 連線數
  CONNECTIONS=$(netstat -an | grep :3838 | wc -l)
  
  # 記錄到日誌
  echo "$(date '+%Y-%m-%d %H:%M:%S') CPU: ${CPU}% MEM: ${MEM}% DISK: ${DISK} CONN: ${CONNECTIONS}" >> /var/log/brandedge/monitor.log
  
  # 警報檢查
  if (( $(echo "$CPU > 80" | bc -l) )); then
    echo "⚠️ CPU 使用率過高: ${CPU}%" | mail -s "BrandEdge Alert" admin@company.com
  fi
  
  sleep 60
done

### 3. 快取管理

```r
# 清理快取
clean_cache <- function(days_old = 7) {
  cache_dir <- "cache/"
  
  if (dir.exists(cache_dir)) {
    files <- list.files(cache_dir, full.names = TRUE, recursive = TRUE)
    file_info <- file.info(files)
    
    # 找出過期檔案
    old_files <- files[difftime(Sys.time(), file_info$mtime, units = "days") > days_old]
    
    # 刪除過期檔案
    if (length(old_files) > 0) {
      unlink(old_files)
      cat(sprintf("已清理 %d 個快取檔案\n", length(old_files)))
    }
  }
}

# 設定定期清理
schedule_cache_cleaning <- function() {
  library(cronR)
  
  cmd <- cron_rscript("clean_cache.R")
  cron_add(cmd, frequency = "daily", at = "02:00")
}
```

## 備份與還原

### 1. 自動備份策略

#### 資料庫備份腳本
```bash
#!/bin/bash
# backup.sh

# 設定變數
BACKUP_DIR="/backup/brandedge"
DB_NAME="brandedge"
DB_USER="brandedge_user"
DATE=$(date +%Y%m%d_%H%M%S)

# 建立備份目錄
mkdir -p $BACKUP_DIR

# 執行備份
pg_dump -U $DB_USER -d $DB_NAME -f "$BACKUP_DIR/db_backup_$DATE.sql"

# 壓縮備份
gzip "$BACKUP_DIR/db_backup_$DATE.sql"

# 保留最近 30 天的備份
find $BACKUP_DIR -name "*.gz" -mtime +30 -delete

# 上傳到雲端（選用）
# aws s3 cp "$BACKUP_DIR/db_backup_$DATE.sql.gz" s3://backup-bucket/brandedge/

echo "備份完成: db_backup_$DATE.sql.gz"
```

#### 應用檔案備份
```bash
#!/bin/bash
# backup_app.sh

APP_DIR="/srv/shiny-server/brandedge"
BACKUP_DIR="/backup/brandedge/app"
DATE=$(date +%Y%m%d_%H%M%S)

# 建立備份
tar -czf "$BACKUP_DIR/app_backup_$DATE.tar.gz" \
    --exclude="$APP_DIR/cache" \
    --exclude="$APP_DIR/logs" \
    --exclude="$APP_DIR/.env" \
    "$APP_DIR"

# 保留最近 7 天的備份
find $BACKUP_DIR -name "*.tar.gz" -mtime +7 -delete
```
